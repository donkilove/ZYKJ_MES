## 1. 项目概述

### 1.1 项目背景

当前 MES 系统（FastAPI + Flutter 架构）需要移植上一版系统（PyQt5）中的生产总页面及其下属多个 Tab 页功能。上一版系统已经实现了完整的生产订单管理、查询、数据统计等功能，并经过充分测试验证。

### 1.2 移植目标

将上一版系统的核心生产管理功能移植到当前 MES 系统，包括：

1. **生产总页面**：生产订单管理的统一入口
2. **Tab 页 1 - 订单管理**：订单创建、编辑、删除、查看
3. **Tab 页 2 - 订单查询**：按角色查询可见订单
4. **Tab 页 3 - 生产数据**：生产进度统计
5. **Tab 页 4 - 品质数据**：质量检验数据
6. **Tab 页 5 - 维修订单**：维修工单管理

### 1.3 核心要求

- **保持订单流转逻辑不变**：严格遵循 `订单流转逻辑详解.md` 中的业务规则
- **支持并行生产模式**：确保上一工序未完成时下一工序可提前开始
- **符合现有架构**：遵循当前 FastAPI + Flutter 的技术架构
- **代码质量**：符合项目编码规范和质量标准

```python
# backend/app/services/order_service.py
class OrderService:
    async def get_orders(
        self, 
        keyword: str = "",
        status: str = "",
        user_role: str = "",
        skip: int = 0,
        limit: int = 50,
    ) -> List[OrderSchema]:
        """获取订单列表"""
        async with get_db_session() as session:
            # 构建查询
            query = select(Order).where(Order.status != '生产完成')
            
            if keyword:
                query = query.where(Order.order_code.like(f"%{keyword}%"))
            
            if status:
                query = query.where(Order.status == status)
            
            # 执行查询
            result = await session.execute(
                query.offset(skip).limit(limit)
            )
            orders = result.scalars().all()
            
            return [OrderSchema.from_orm(o) for o in orders]
    
    async def create_order(
        self, 
        order_data: OrderCreateSchema,
        processes: List[ProcessConfig],
        operator: str,
    ) -> OrderSchema:
        """创建订单并初始化所有工序"""
        async with get_db_session() as session:
            async with session.begin():
                # 1. 创建订单主记录
                order = Order(
                    order_code=order_data.order_code,
                    product_id=order_data.product_id,
                    quantity=order_data.quantity,
                    status='待生产',
                    start_date=order_data.start_date,
                    end_date=order_data.delivery_date,
                    operator=operator,
                )
                session.add(order)
                await session.flush()
                
                # 2. 创建工序记录
                for i, process in enumerate(processes):
                    visible_qty = order_data.quantity if i == 0 else 0
                    order_process = OrderProcess(
                        order_id=order.order_id,
                        process_name=process.name,
                        status='待生产',
                        operator=process.operator_role,
                        process_order=i + 1,
                        visible_quantity=visible_qty,
                    )
                    session.add(order_process)
                    
                    # 3. 创建工序状态记录
                    process_status = ProcessStatus(
                        order_id=order.order_id,
                        process_name=process.name,
                        status='待生产',
                        product_quantity=0,
                    )
                    session.add(process_status)
                
                await session.commit()
                return OrderSchema.from_orm(order)
```

#### 3.3.1 功能清单

#### 3.3.2 业务逻辑

严格遵循 `订单流转逻辑详解.md` 中的规则：

1. **可见性判定**：根据角色和工序状态判断订单是否可见
2. **数量显示**：第一道工序显示订单总量，其他工序显示 visible_quantity
3. **并行生产**：支持多个工序同时生产
4. **消息通知**：订单流转时自动通知下一工序

#### 3.3.3 关键代码

```python
# backend/app/services/order_service.py
class OrderService:
    async def get_orders_by_role(
        self, 
        user_role: str,
        keyword: str = "",
    ) -> List[OrderWithProcessSchema]:
        """根据角色获取可见订单列表"""
        async with get_db_session() as session:
            # 1. 查询所有非"生产完成"状态的订单
            query = select(Order).where(Order.status != '生产完成')
            
            if keyword:
                query = query.where(Order.order_code.like(f"%{keyword}%"))
            
            result = await session.execute(query)
            orders = result.scalars().all()
            
            visible_orders = []
            
            for order in orders:
                # 2. 获取订单工序列表
                processes_query = select(OrderProcess).where(
                    OrderProcess.order_id == order.order_id
                ).order_by(OrderProcess.process_order)
                
                processes_result = await session.execute(processes_query)
                processes = processes_result.scalars().all()
                
                # 3. 根据订单状态判断可见性
                is_visible, visible_process, visible_quantity = \
                    self._check_order_visibility(order, processes, user_role)
                
                if is_visible and visible_process:
                    # 4. 设置可见数量
                    if visible_quantity > 0:
                        order.quantity = visible_quantity
                    
                    visible_orders.append(
                        OrderWithProcessSchema.from_orm(order, process=visible_process)
                    )
            
            return visible_orders
    
    def _check_order_visibility(
        self, 
        order: Order, 
        processes: List[OrderProcess],
        user_role: str,
    ) -> Tuple[bool, Optional[OrderProcess], int]:
        """检查订单对某角色是否可见"""
        if order.status == '待生产':
            # 待生产状态：检查所有待生产的工序
            for i, process in enumerate(processes):
                if process.operator == user_role:
                    if process.status == '待生产':
                        # 计算可见数量
                        if process.process_order == 1:
                            visible_quantity = order.quantity
                        else:
                            visible_quantity = process.visible_quantity
                        
                        if visible_quantity > 0:
                            return True, process, visible_quantity
                    
                    elif process.status == '生产中':
                        # 工序处于生产中状态（可再次首件），可见
                        return True, process, order.quantity
        
        elif order.status == '生产中':
            # 生产中状态：检查所有进行中或生产中的工序
            for process in processes:
                if process.operator == user_role:
                    if process.status in ['进行中', '生产中']:
                        return True, process, order.quantity
                    elif process.status in ['生产完成', '已完成']:
                        # 当前角色的工序已完成，隐藏订单
                        return False, None, 0
        
        return False, None, 0
```

```python
# backend/app/services/statistics_service.py
class StatisticsService:
    async def get_order_progress(self, order_id: str) -> OrderProgressSchema:
        """获取订单进度统计"""
        async with get_db_session() as session:
            # 获取订单信息
            order = await session.get(Order, order_id)
            
            # 获取所有工序
            processes = await session.execute(
                select(OrderProcess).where(
                    OrderProcess.order_id == order_id
                ).order_by(OrderProcess.process_order)
            )
            processes = processes.scalars().all()
            
            # 获取各工序的生产数量
            status_list = await session.execute(
                select(ProcessStatus).where(
                    ProcessStatus.order_id == order_id
                )
            )
            status_list = status_list.scalars().all()
            
            # 计算进度
            process_stats = []
            total_completed = 0
            
            for process in processes:
                status = next(
                    (s for s in status_list if s.process_name == process.process_name),
                    None
                )
                
                completed = status.product_quantity if status else 0
                progress = (completed / order.quantity * 100) if order.quantity > 0 else 0
                
                process_stats.append(ProcessStatSchema(
                    process_name=process.process_name,
                    status=process.status,
                    completed=completed,
                    progress=progress,
                ))
                
                if process.status in ['生产完成', '已完成']:
                    total_completed += 1
            
            overall_progress = (
                total_completed / len(processes) * 100 
                if processes else 0
            )
            
            return OrderProgressSchema(
                order_id=order_id,
                total_quantity=order.quantity,
                processes=process_stats,
                overall_progress=overall_progress,
            )
```

```sql
CREATE TABLE orders (
```

```python
from sqlalchemy import Column, String, Integer, DateTime, Text, ForeignKey
from sqlalchemy.orm import relationship
from datetime import datetime
from .base import Base

class Order(Base):
    __tablename__ = "orders"
    
    order_id = Column(String(50), primary_key=True)
    order_code = Column(String(50), nullable=False)
    product_id = Column(Integer, ForeignKey("products.id"), nullable=False)
    quantity = Column(Integer, nullable=False, default=0)
    produced_quantity = Column(Integer, nullable=False, default=0)
    status = Column(String(50), nullable=False, default='待生产')
    current_process = Column(String(100))
    start_date = Column(DateTime)
    end_date = Column(DateTime)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    remark = Column(Text)
    operator = Column(String(100))
    
    # 关系
    processes = relationship("OrderProcess", back_populates="order", cascade="all, delete-orphan")
    process_status_list = relationship("ProcessStatus", back_populates="order", cascade="all, delete-orphan")
    production_records = relationship("ProductionRecord", back_populates="order")
    messages = relationship("Message", back_populates="order")

class OrderProcess(Base):
    __tablename__ = "order_processes"
    
    id = Column(Integer, primary_key=True)
    order_id = Column(String(50), ForeignKey("orders.order_id"), nullable=False)
    process_name = Column(String(100), nullable=False)
    status = Column(String(50), nullable=False, default='待生产')
    operator = Column(String(100), nullable=False)
    process_order = Column(Integer, nullable=False, default=0)
    visible_quantity = Column(Integer, nullable=False, default=0)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    
    # 关系
    order = relationship("Order", back_populates="processes")

class ProcessStatus(Base):
    __tablename__ = "process_status"
    
    id = Column(Integer, primary_key=True)
    order_id = Column(String(50), ForeignKey("orders.order_id"), nullable=False)
    process_name = Column(String(100), nullable=False)
    status = Column(String(50), nullable=False, default='待生产')
    product_quantity = Column(Integer, nullable=False, default=0)
    ongoing_users = Column(Text, default='[]')
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
    
    # 关系
    order = relationship("Order", back_populates="process_status_list")

class ProductionRecord(Base):
    __tablename__ = "production_records"
    
    id = Column(Integer, primary_key=True)
    order_id = Column(String(50), ForeignKey("orders.order_id"), nullable=False)
    process_name = Column(String(100), nullable=False)
    operator = Column(String(100), nullable=False)
    production_quantity = Column(Integer, nullable=False, default=0)
    record_type = Column(String(50), nullable=False, default='正常生产')
    created_at = Column(DateTime, default=datetime.utcnow)
    
    # 关系
    order = relationship("Order", back_populates="production_records")

class Message(Base):
    __tablename__ = "messages"
    
    id = Column(Integer, primary_key=True)
    sender = Column(String(100), nullable=False)
    recipient = Column(String(100), nullable=False)
    title = Column(String(200), nullable=False)
    content = Column(Text, nullable=False)
    message_type = Column(String(50), nullable=False, default='system')
    related_order_id = Column(String(50), ForeignKey("orders.order_id"))
    is_read = Column(Boolean, nullable=False, default=False)
    created_at = Column(DateTime, default=datetime.utcnow)
    
    # 关系
    order = relationship("Order", back_populates="messages")
```
